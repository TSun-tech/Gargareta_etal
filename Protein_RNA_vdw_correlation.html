# Main plot
pmain <- ggscatter(protein_rna, x = "human_RNA", y = "human_MSEvdw", font.label = "italic",
               color = "skyblue3", size = 2,label = "label", repel = T,
              add = "reg.line",
              add.params = list(color = "#00AFBB", fill = "lightgray"),
              conf.int = T, cor.coef = T,
               cor.coef.args =list(method = "pearson", label.sep = "\n"))
# Marginal densities along x axis
xdens <- axis_canvas(pmain, axis = "x")+
  geom_density(data = protein_rna, aes(x = human_RNA),
              alpha = 0.7, size = 0.2, color = "skyblue3", fill = "skyblue3")
# Marginal densities along y axis
# Need to set coord_flip = TRUE, if you plan to use coord_flip()
ydens <- axis_canvas(pmain, axis = "y", coord_flip = TRUE)+
  geom_density(data = protein_rna, aes(x = human_MSEvdw),
                alpha = 0.7, size = 0.2, color = "skyblue3", fill = "skyblue3")+
  coord_flip()
p1 <- insert_xaxis_grob(pmain, xdens, grid::unit(.2, "null"), position = "top")
p2<- insert_yaxis_grob(p1, ydens, grid::unit(.2, "null"), position = "right")
ggdraw(p2)

pdf(".../2021_Human_MSE_RNA_vdw_correlation.pdf")
print(ggdraw(p2))
dev.off()
